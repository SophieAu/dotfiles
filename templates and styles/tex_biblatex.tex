% Personal favourite options:
\usepackage[style=numeric,sorting=nyt]{biblatex}



%Per default, \printbibliography gives the bibliography the title "References", you can change it though
\printbibliography[title={Custom Header}]

% you can even only print entries that match a certain keyword in their keywords field
\printbibliography[keyword=boring,title={Only Boring Papers}]
% or only entries of a certain type
\printbibliography[type=article,title={Articles only}]

% To add the bibliography to the TOC, add the following option:
\printbibliography[heading=bibintoc] % Adds as an unnumbered chapter/section (if no chapters allowed)
\printbibliography[heading=subbibintoc] % Adds as an unnumbered section/subsection (depending on bibintoc level)
